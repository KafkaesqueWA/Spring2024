\section*{Problem 1.1.8}
Write the number in the form $a + bi$
\begin{align*}
	\frac{(8 + 2i) - (1 - i)}{(2 + i)^2} &= \frac{7 + 3i}{3 + 4i} \\
					     &= \left(\frac{7 + 3i}{3 + 4i}\right)\left(\frac{3 - 4i}{3 - 4i}\right) \\ 
					     &= \frac{33 - 19i}{25} \\
					     &= \frac{33}{25} - \frac{19}{25}i
\end{align*}
